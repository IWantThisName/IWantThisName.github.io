Elasticsearch的安装启动

我是在windows下进行安装的,网上下载下面这个文件

解压 elasticsearch-5.2.2 后

点击elasticsearch.bat,即启动了elasticsearch

可以在浏览器中输入http://localhost:9200 查看elasticsearch的一些信息

需要注意的是,5.2.2版本需要jdk1.8环境

添加索引

如果index/type/id已经存在,则覆盖
否则为添加 

public class ElasticsearchTest {

    private TransportClient client;

    @Before
    public void init(){
        try {
            //初始化client对象
            client = new PreBuiltTransportClient(Settings.EMPTY).addTransportAddress(new InetSocketTransportAddress(InetAddress
                    .getByName("localhost"),9300));
        } catch (UnknownHostException e) {
            e.printStackTrace();
        }
    }

    @Test
    public void testInsert() throws IOException {
        //创建一个json字符串
        XContentBuilder builder = XContentFactory.jsonBuilder().startObject()
                .field("name","华为").field("price",1000).field("desc","好用").endObject();
        //添加索引,prepareIndex的三个参数分表代表Elasticsearch中的 index,type和id
        //在ElasticSearch中 index相当于数据库中的database,type相当于table,id相当于某一行
        //上面构造的json字符串就代表具体的那一行信息
        IndexResponse indexResponse = client.prepareIndex("shopp", "phone", "1").setSource(builder.string()).get();
        //打印结果信息
        System.out.println(indexResponse.getIndex());
        System.out.println(indexResponse.getType());
        System.out.println(indexResponse.getId());
        System.out.println(indexResponse.getVersion());
    }
}

一次查询多个结果

@Test
public void testMultiGet(){
    MultiGetResponse responses = client.prepareMultiGet().add("shopp", "phone", "1")
            .add("shopp", "phone", "2").get();
    for (MultiGetItemResponse response : responses) {
        GetResponse response1 = response.getResponse();
        if (response1.isExists()) {
            Map<String, Object> source = response1.getSource();
            System.out.println(response1.getId());
            for (String s : source.keySet()) {
                System.out.println(s + " : " + source.get(s));
            }
            System.out.println("------------");
        }
    }
}

BulkAPI 一次请求进行多次操作

@Test
public void testBulk() throws IOException {
    BulkRequestBuilder request = client.prepareBulk();
    //加了一次添加操作
    request.add(client.prepareIndex("shopp","phone","3").setSource(XContentFactory.jsonBuilder().startObject()
            .field("name","vivo").field("price",2000).field("desc","6666").endObject()));
    //加一次删除操作
    request.add(client.prepareDelete("shopp","phone","1"));
    //发送请求
    BulkResponse responses = request.get();
    if(responses.hasFailures()){
        System.out.println(responses.buildFailureMessage());
    }
}

scroll查询

@Test
public void testScrollQuery() throws IOException {
    //设置查询条件
    TermQueryBuilder termQuery = QueryBuilders.termQuery("name", "华为");
    //设置查询的一些参数,查询的index,查询的数目,查询排序
    SearchRequestBuilder builder = client.prepareSearch("ttt").setQuery(termQuery)
            .setSize(100).setScroll(new TimeValue(6000)).addSort(FieldSortBuilder.DOC_FIELD_NAME, SortOrder.ASC);
    //新建一个高亮建造器
    HighlightBuilder highlightBuilder = new HighlightBuilder();
    //设置高亮的标签和高亮的域
    highlightBuilder.preTags("<em>").postTags("</em>").field("name");
    //将高亮加入到查询中
    builder.highlighter(highlightBuilder);
    //获取查询结果
    SearchResponse response = builder.get();
    //获取结果值
    SearchHits hits = response.getHits();
    SearchHit[] searchHits = hits.getHits();
    ObjectMapper mapper = new ObjectMapper();
    //遍历结果值
    for (SearchHit searchHit : searchHits) {
        //取高亮数据
        Map<String, HighlightField> fields = searchHit.highlightFields();
        //获取高亮数据中指定域的值
        HighlightField field = fields.get("name");
        //获取具体的标签和值然后进行拼接
        Text[] fragments = field.fragments();
        StringBuilder sb1 = new StringBuilder();
        for (Text fragment : fragments) {
            sb1.append(fragment);
        }
        String name = sb1.toString();
        System.out.println(name);
        //将文档对象转正json值
        String phoneJson = searchHit.getSourceAsString();
        Phone phone = mapper.readValue(phoneJson, Phone.class);
        phone.setName(name);
        System.out.println(phone);
    }
}

创建索引时设置mapping 指定分析器

//创建索引结构
XContentBuilder xbuilder = XContentFactory
        .jsonBuilder()
        .startObject()
        .startObject("computer")
        .startObject("properties")
        .startObject("name")
        .field("type", "string")
        .field("store", "yes")
        .field("analyzer", "ik_max_word")
        .field("index", "analyzed")
        .endObject()
        .startObject("price")
        .field("type", "long")
        .field("store", "no")
        .field("index", "not_analyzed")
        .endObject()
        .startObject("desc")
        .field("type", "string")
        .field("store", "yes")
        .field("analyzer", "ik_max_word")
        .field("index", "analyzed")
        .endObject()
        .endObject()
        .endObject()
        .endObject();
PutMappingRequest mapping = Requests.putMappingRequest("ttt").type("computer").source(xbuilder);
//创建索引
client.admin().indices().prepareCreate("ttt").execute().actionGet();
client.admin().indices().putMapping(mapping).actionGet();
